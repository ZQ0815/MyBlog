# 继承父类的类在初始化的时候的细节问题
```
# 接口的定义
@interface Volume : NSObject
{
    int val;
    int min,max,step;
}
- (id)initWithMin:(int) a max:(int) b step:(int) s;
- (int) value;
- (id) up;
- (id) down;
@end

# 接口的实现
@implementation Volume
- (id)initWithMin:(int) a max:(int) b step:(int) s
{
    self = [super init]; 
    if (self != nil ){
        val = min = a;
        max = b;
        step = s;
    }
}

- (int) value
{
    return value;
}

- (id) up
{
    if((val += step) > max)
        val = max;
    return self;
}

- (id) down
{
    if((val -= step) < max)
        val = min;
    return self;
}
@end
```
